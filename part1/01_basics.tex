\begin{frame}{\\Сущность понятия технологии}
	\topline
	\justifying
	\large
	\textit{Искусство} -- результат или есть или нет, результат не гарантирован.
	
	\bigskip
	
	\textit{Технология} -- возникает в ходе человеческой деятельности, \uline{результат гарантирован}.
	
	\bigskip
	
	\textit{Развитая технология} -- целостная, целесообразная, регламентированная, проведена стандартизация и унификация процессов и средств.
\end{frame}
