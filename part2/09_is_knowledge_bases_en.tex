\begin{frame}{\\Typology of signs}
	\topline
	\justifying
	\small
	
	\vspace{10mm}
	Signs used in different languages are characterized by:
	\begin{textitemize}
		\item syntactic structure, by coincidence (isomorphism)
		which for different signs their synonymy is assumed;
		\item denotational semantics, i.e. the essence that
		is designated by the corresponding sign;
		\item type (class) of the designated entity, which can
		be:
		\begin{textitemize}
			\item material (physical) element (point)
			an abstract space, a set, which can be a connection, a class, or a structure (information construct)
			\item real and fictional entity;
			\item constant (specific) and variable
			(arbitrary) entity;
			\item permanently existing and temporarily existing
			entity (past, present, future);
		\end{textitemize}
		\item ...
	\end{textitemize}
	
\end{frame}

\begin{frame}{\\The structure of the sign --- Frege's triangle}
	\topline
	\justifying
	
	\textbf{I. Thing}, object, phenomenon of reality, etc. Another name is --- \textbf{denotation}
	
	\vspace{0.5em}
	
	\textbf{II. Sign}: in linguistics, for example, a phonetic word or a written word in mathematics --- a mathematical symbol; another name, adopted especially in philosophy and mathematical logic, --- \textbf{name}.
	
	\vspace{0.5em}
	
	\textbf{III. Concept} of an object or thing. Other names: in linguistics --- significator, designatum; in mathematics --- the sense of a name, or the concept of a denotate.
	
\end{frame}

\begin{frame}{\\Denotation}
	\topline
	\justifying
	\small
	
	This relation is used when it is necessary to describe the denotational semantics of another language using one language.
	
	\bigskip
	
	In fact, we are talking about the translation of a given sign, which is part of a certain text under consideration, belonging to a certain language under study (object language), into a certain metalanguage, the denotational semantics of which is considered to be known to us a priori.
	
	\bigskip
	
	The specified translation is a connection between a given sign and a synonymous sign that is part of a text belonging to the specified metalanguage.
	
\end{frame}

\begin{frame}{\\Formal Logical Models}

	\begin{textitemize}
		\item Propositional Calculus
		\item Predicate Calculus
		\item Fuzzy logic
	\end{textitemize}
	
	It is based on a formal theory (system):
	\begin{textitemize}
		\item T --- set of terms, alphabet of the system
		\item P --- syntactic rules for constructing expressions from basic terms

	\end{textitemize}

\begin{frame}{\\Propositional Calculus}
	\topline
	\justifying
	
	А \textbf{statement} is an indivisible, grammatically correct sentence that can be signized as true or false.
	
	\vspace{1em}
	
	А \textbf{complex statement} is a combination of simple statements using logical connectives.
	
	\vspace{1em}
	
	Example:
	\begin{textitemize}
		\item A --- The Moon is the Earth's satellite
		\item B --- The Sun is the Earth's satellite
		\item A \& B --- false
	\end{textitemize}
	
\end{frame}

\begin{frame}{\\Predicate Calculus}
	\topline
	\justifying
	
	\textbf{Predicate calculus} allows one to model a subject domain and test various hypotheses regarding this subject domain using the developed predicate system.
	
	\vspace{1em}
	
	\textbf{Predicate} --- a function on the set M=M1*M2*…*Mn, taking the value true or false.
	
	\vspace{1em}
	
	Example:
	\begin{textitemize}
		\item If a philosopher wins a race against someone, that person will admire him.
		\item (any X) (any Y) (PHILOSOPHER(X) \textasciicircum BEATS(X, Y) -> ADMIRE(Y, X)).
	\end{textitemize}
	
\end{frame}

\begin{frame}{\\Logical Models: An Example}
	\topline
	\justifying
	
	\[
	\forall x \forall a \forall b \forall c \Bigl(
	RT(x) \land K(x,a) \land K(x,b) \land H(x,c)
	\rightarrow
	Pyth\bigl(a,b,c\bigr)
	\Bigr).
	\]
	
	\[
	\forall a \forall b \forall c \Bigl(
	Pyth(a,b,c) \leftrightarrow
	\exists u \exists v \exists w \bigl(
	S(a,u) \land S(b,v) \land S(c,w) \land Plus(u,v,w)
	\bigr)
	\Bigr).
	\]
	
\end{frame}

\begin{frame}{\\Logical Models: An Example}
	\topline
	\justifying
	
	\begin{flalign*}
		RT(x) &:\; x \text{ --- right triangle.} \\
		K(x, a) &:\; a \text{ --- leg of triangle } x. \\
		H(x, c) &:\; c \text{ --- hypotenuse of triangle } x. \\
		S(a, u) &:\; u \text{ --- the square of the length of the segment} a. \\
		Plus(u, v, w) &:\; w \text{ is the sum of } u \text{ and } v. \\
		Pyth(a, b, c) &:\; \text{the square of the hypotenuse is equal to} \\
		&\text{ sum of the squares of the legs for } a, b, c.
	\end{flalign*}
	
\end{frame}

\begin{frame}{\\Production Models}
	\topline
	\justifying
	
	\textbf{A production model} is a set of productions --- rules of the form "If"{} --- "Then"{};
	
	Formally, the product is described as:
	\begin{textitemize}
		\item W --- scope of application of the product.
		\item U --- precondition --- (truth of production).
		\item P --- conditions of use of the product.
		\item A->B --- core of the production, rule of the type If …, then … .
		\item C --- production postcondition, actions after processing the product.
	\end{textitemize}
	
	\textbf{Production system} = fact base + production + interpreter.
	
\end{frame}

\begin{frame}{\\Homonymy}

\begin{frame}{\\Polysemy}
	\topline
	\justifying
	
	\vspace{10mm}
	The presence of different senses for the same symbol. This term is typically used in situations where these different senses are somehow related (as opposed to homonymy).
	
	\vspace{10mm}
	\textbf{Example}: definitions of the concept for the phrase "Major repairs"{},
	encountered at one of the domestic enterprises:
	\begin{textitemize}
		\item work on technical maintenance and repair of equipment,
		performed during a major shutdown, i.e. a stop of production
		lines lasting more than 24 hours
		\item repairs performed to restore serviceability and complete
		or close to full restoration of the product's resource with replacement
		or restoration of any of its parts, including the basic ones
	\end{textitemize}
	
\end{frame}

\begin{frame}{\\Synonymy}
	\topline
	\justifying
	
	Indicates the equivalence, but not the identity, of the signs.
	
	\vspace{1em}
	
	Equivalence means:
	\begin{textitemize}
		\item or correlation with the same denotation (subject);
		\item or correlation with the same concept, or more precisely with that part of it that contains signizing information.
	\end{textitemize}
	
\end{frame}

\begin{frame}{\\Ontology}
	\topline
	\justifying
	
	Ontology (from the ancient Greek ontos --- being, logos --- teaching, concept) --- a term defining the teaching about being, existence, in contrast to epistemology --- the teaching about knowledge.
	
	\vspace{1em}
	
	In the philosophical sense (and this term is borrowed from philosophy), ontology is a certain system of categories that are a consequence of certain views of the world.
	
\end{frame}

\begin{frame}{\\History of the concept}
	\topline
	\justifying
	
	\vspace{10mm}
	
	Currently, the term “ontology” has moved into the field of information technology, where it has been used by a number of research communities in artificial intelligence, first in the field of knowledge engineering, in natural language processing, and then in knowledge representation [Gruber, 1993; Uschold, Jasper, 1999].
	
	\vspace{0.5em}
	
	In the late 1990s and early 2000s, the concept of ontology also became widely used in domains such as information mining, web retrieval, and knowledge management [Fensel, 2001; Gomez-Perez et al, 2006]
	
\end{frame}

\begin{frame}{\\History of the concept (continued)}
	\topline
	\justifying
	
	Later, ontologies began to be considered as a key element in the Semantic Web project -- a new stage in the development of the WWW (Word Wide Web).
	
	\vspace{1em}
	
	If the existing Web is a huge set of documents that are linked by cross-references, then the emerging Semantic Web should add to the existing network a set of ontologies and meta-descriptions of knowledge contained in Web documents (including standards and software tools) [BernersLee et al, 2001; Domingue et al, 2011].
	
\end{frame}

\begin{frame}{\\The concept of ontology}
	\topline
	\justifying
	
	\vspace{2em}
	In general, an ontology consists of a hierarchy of concepts in a subject domain, the connections between them, and the laws that operate within the framework of this model.
	
	\vspace{0.5em}
	An ontology is constructed as a network consisting of concepts, the relationships between them, and descriptions of the properties of these concepts. Relationships can be of various types (e.g., "is," "consists of," "is an executor," etc.).
	
	\vspace{0.5em}
	To fulfill the role of a common language, an ontology includes a dictionary (a set of words/lexical units) and a set of definitions of concepts (formal, described, for example, in some logical language, or informal), defining the sense of the elements of the dictionary.
	
	\vspace{0.5em}
	The connections between vocabulary (signs) and semantics limit the number of possible interpretations of signs.
	
\end{frame}

\begin{frame}{\\More strictly}
	\topline
	\justifying
	
	\begin{textitemize}
		\item \textbf{Ontology} is an explicit formal specification of the conceptualization of some subject domain, shared by some community of agents
		\begin{textitemize}
			\item In other words, a formal (strict) description of a system of concepts of a certain subject domain, agreed upon within a certain community
			\item The purpose of developing ontologies is to solve problems such as synonymy, homonymy, and polysemy
		\end{textitemize}
	\end{textitemize}
	
\end{frame}

\begin{frame}{Even more strict:\\formal model of ontology}
	\topline
	\justifying
	
	O = < C, P, R, A >,
	
	\vspace{1em}
	
	Where:
	\begin{textitemize}
		\item C --- a finite set of concepts (entity classes) of a subject domain;
		\item P --- a finite set of properties of these concepts (classes);
		\item R --- a finite set of relationships between concepts (classes);
		\item A --- a set of axioms, statements constructed from these concepts, their properties and connections between them.
	\end{textitemize}
	
\end{frame}

\begin{frame}{\\Example of disambiguation}
	\topline
	\justifying
	
	After establishing the connection between the word "Jaguar"{} and the intensional definition "a predatory animal of the cat family"{}, we exclude cars and drinks from the set of possible interpretations of this word.
	
	\vspace{1em}
	As a result, ontologies allow us to correctly correlate the signs used by people and computer systems with concepts (semantic models) and designated objects of the real world.
	
\end{frame}

\begin{frame}{\\Formal taxonomies}
	\topline

		\item for the INSTANCE-CLASS relationship, the so-called "inheritance"{} holds: if A is a subclass of class B, then every instance of class A is also an instance of class B
	\end{textitemize}

\begin{frame}{Descriptive Logic (DL)\\Formal Framework of SW}
	\topline
	\justifying
	A family of formal knowledge representation languages:
	\begin{textitemize}
		\item Balance between expressive power and output complexity
		\item More expressive than propositional logics
		\item Less expressive than first-order predicate logic
		\item Usually solvable with efficient inference procedures
	\end{textitemize}
\end{frame}

\begin{frame}{DL Basic Concepts\\Concepts and Roles}
	\topline
	\justifying
	
	\begin{textitemize}
		\item \textbf{Concept} --- unary predicate (class, set)
		\item \textbf{Role} --- two-place predicate (binary relation)
		\item \textbf{TBox} --- a set of terminological axioms
		\item \textbf{ABox} --- a set of statements about individuals
	\end{textitemize}
\end{frame}

\begin{frame}{Knowledge Base in DL\\Representation Structure}
	\topline
	\justifying
	
	Knowledge Base = TBox + ABox
	
	\textbf{Terminological axioms (TBox)}:
	\begin{textitemize}
		\item Nesting Axiom: $C \sqsubseteq D$
		\item Axiom of Equivalence: $C \equiv D$
	\end{textitemize}
	
	\textbf{Statements about individuals (ABox):}
	\begin{textitemize}
		\item Concept Belonging: $C(a)$ or $a:C$
		\item Role relationship: $R(a,b)$ or $aRb$
	\end{textitemize}
\end{frame}

\begin{frame}{Example Terminology (TBox)\\Formal Representation}
	\topline
	\justifying
	\begin{textitemize}
		\item Woman $\equiv$ Person $\sqcap$ Female
		\item Mother $\equiv$ Woman $\sqcap$ $\exists$ hasChild.$\top$
		\item $\forall$ hasChild.Person $\sqsubseteq$ Person
		\item Doctor $\sqsubseteq$ Person
	\end{textitemize}
	
	\vspace{0.5cm}
	Interpretation: A woman is a female person; A mother is a woman who has a child; Every doctor is a Person
\end{frame}

		\item \textbf{IRI} --- Unicode character support

\end{frame}

\begin{frame}{Properties of OWL Individuals\\Two Kinds of Properties}

\begin{frame}{Unification of information representation\\in computer system memory}
	\topline
	\justifying
	
	Unification of information presentation involves:
	\begin{textitemize}
		\item \textbf{Syntactic unification} of information\\
		(unification of the form of presentation and coding of information).
		\item \textbf{Semantic unification} of information\\

	\end{textitemize}

\begin{frame}{Semantic representation of information\\Main idea}
	\topline
	\justifying
	
	\textbf{Semantic representation of information} ---
	a symbolic construction (text), the syntactic structure of which is close to its sense, that is, the configuration of connections between the described entities.
	
	\medskip
	The main principle: as little as possible of the superfluous, irrelevant to the sense of the information being presented.
	
	\medskip
	A semantic representation is an abstract symbolic construction that is an \uline{invariant} of the maximal class of semantically equivalent symbolic constructions.
	
\end{frame}


\begin{frame}{Semantic representation of information\\Content structure}
	\topline
	\justifying
	
	The semantic representation of any information comes down to:
	\begin{textitemize}
		\item to the list of signs of the described entities (primary and secondary, which are fragments of the structure);
		\item to an explicit description of the connections between the signs of secondary entities and the corresponding fragments;
		\item to the description of other relationships between the described entities.
	\end{textitemize}
	
\end{frame}


\begin{frame}{Standardization of semantic representation\\and knowledge compatibility}
	\topline
	\justifying
	
	Competent unification of the semantic presentation of information:
	\begin{textitemize}
		\item should not limit the creative freedom of authors of scientific and technical knowledge and developers of knowledge bases;
		\item must guarantee \textit{semantic compatibility} of knowledge presented by different authors (subject to compliance with the rules of knowledge construction);
		\item must constantly evolve, the current version of the standard is not a dogma, but a support for further improvement.
	\end{textitemize}
	
\end{frame}


\begin{frame}{\\The concept of "sense"}
	\topline
	\justifying
	
	\textbf{sense} --- an abstract symbolic construction of the internal language of a computer system, which is an \textbf{invariant} of the maximal class of semantically equivalent texts in different languages.
	
	\medskip
	Requirements for sense representation:
	\begin{textitemize}
		\item \textbf{Universality}: the ability to present any information.
		\item \textbf{Absence of synonymy of signs}: prohibition of multiple occurrences of signs with the same denotations.
		\item \textbf{Absence of duplication of information} in the form of semantically equivalent texts.
	\end{textitemize}
	
\end{frame}


\begin{frame}{Requirements for sense\\(continued)}
	\topline
	\justifying
	
	\begin{textitemize}
		\item \textbf{Absence of homonymous signs} (including pronouns).
		\item \textbf{Lack of internal structure of characters}\\
		(atomic nature of signs).
		\item \textbf{Absence of inflection (sign forms)}\\
		as a consequence of the lack of internal structure of signs.
		\item \textbf{Absence of fragments} that are not signs\\
		(separators, delimiters, etc.).
		\item \textbf{Separation of connection signs}, connecting any signs with incidence relations.
	\end{textitemize}
	
\end{frame}


\begin{frame}{Nonlinearity of semantic representation\\of information}
	\topline
	\justifying
	
	The nonlinearity of the semantic representation is due to the fact that:
	\begin{textitemize}
		\item each described entity can have an unlimited number of connections with other entities;
		\item each entity has exactly one sign (synonymy of signs is prohibited);
		\item all connections between entities are described by connections between their signs.
	\end{textitemize}
	
\end{frame}


\begin{frame}{\\Language of semantic representation}
	\topline
	\justifying
	
	The key property of the language of semantic representation of information:
	\begin{textitemize}
		\item \textbf{Unambiguity of information representation} in the memory of each computer system.
	\end{textitemize}
	
	\medskip
	This means the absence of semantically equivalent symbolic constructions belonging to the semantic language and stored in the same semantic memory (with logical equivalence allowed for compactness).
	
\end{frame}


\begin{frame}{Refined semantic network:\\main principle}
	\topline
	\justifying
	
	\textbf{Refined semantic network} ---
	\uline{extremely} compact semantic information model of the system of entities under consideration.
	
	\medskip
	The main principle: absolutely nothing superfluous, not related to the sense of the information presented.
	
	\medskip
	The refined semantic network is used as a \textit{internal consumption semantic network} for the semantic representation of information in the internal memory of a computer system.
	
\end{frame}


\begin{frame}{Refined semantic network:\\clarification of principles}
	\topline
	\justifying
	
	The clarification of the principles of semantic representation of information in a refined semantic network is based on:
	\begin{textitemize}
		\item a clear opposition between \textit{internal language} (for storing information) and \textit{external languages} (for communication with users and other systems);
		\item maximum simplification of the syntax of the internal language while maintaining universality;
		\item exclusion from the internal language of means oriented towards communication (conjunctions, prepositions, separators, declensions, conjugations, etc.).
	\end{textitemize}
	
\end{frame}


\begin{frame}{Refined semantic network:\\fragments and incidence}
	\topline
	\justifying
	
	\footnotesize
	
	\textbf{Principle 1}:
	\begin{textitemize}
		\item Each fragment of the \textit{refined semantic network} is either a sign (node, edge, arc) or a set of signs connected by an incidence relation.
		\item Incidence is a binary oriented relation that connects the signs of connections with the signs of entities or with the entities themselves.
	\end{textitemize}
	
	\medskip
	The structuring of a refined semantic network is described explicitly by:
	\begin{textitemize}
		\item nodes denoting non-elementary fragments (sets of nodes, edges, arcs);
		\item membership arcs connecting such nodes with elements of sets;
		\item relationships between fragments and other entities, as well as fragment classes.
	\end{textitemize}
	
\end{frame}


\begin{frame}{Refined semantic network:\\atomicity of signs}
	\topline
	\justifying
	
	\textbf{Principle 2}:
	\begin{textitemize}
		\item All signs of the refined semantic network are syntactically elementary (atomic), the internal structure of signs does not play a role in semantic analysis.
		\item A set of signs corresponds to an alphabet that defines the syntactic typology of elementary fragments and has a semantic interpretation.
	\end{textitemize}
	
	\medskip
	The signs of the refined semantic network are not names (terms) and are not tied to natural languages, which eliminates terminological disputes.
	
\end{frame}


\begin{frame}{Refined semantic network:\\atomicity of signs}
	\topline
	\justifying
	
	A consequence of atomicity:
	\begin{textitemize}
		\item It is not necessary to use fragments that are not entity signs (letters, separators, limiters, etc.).
		\item There is no need to contrast simple signs and expressions: both correspond to network elements that have certain denotations.
		\item Expressions additionally correspond to non-elementary network fragments that specify the entities they designate.
	\end{textitemize}
	
\end{frame}


\begin{frame}{Refined semantic network:\\Explicit Representation of Relationships}
	\topline
	\justifying
	
	\textbf{Principle 3}:
	\begin{textitemize}
		\item All connections between the described entities are represented explicitly by signs of the corresponding connections, incident to the signs of the entities.
		\item For binary connections, the connection signs are the edges or arcs of the network.
	\end{textitemize}
	
	\medskip
	Consequence: there is no need for declensions, conjugations, genders, semantically loaded sequences of words and other linguistic means.
	
\end{frame}


\begin{frame}{Refined semantic network:\\absence of synonymy}
	\topline
	\justifying
	
	\textbf{Principle 4}:
	\begin{textitemize}
		\item All signs enter the network once; pairs of synonymous signs (with identical denotations) are absent.
		\item Different elements of the network are a priori considered to be signs of different entities, even if they belong to the same syntactic type.
	\end{textitemize}
	
	\medskip
	The appearance of synonymous signs makes the network incorrect and requires their identification (gluing).
	
\end{frame}


\begin{frame}{Refined semantic network:\\absence of homonymy and duplication}
	\topline
	\justifying
	
	\textbf{Principle 5}:
	\begin{textitemize}
		\item There are no homonymous signs on the network, that is, signs with several denotations.
		\item The network does not contain duplication of information in the form of semantically equivalent fragments of refined semantic networks.
	\end{textitemize}
	
	\medskip
	Semantically equivalent networks are isomorphic networks whose correspondence pairs link synonymous signs; the exclusion of homonymy and synonymy excludes the semantic equivalence of fragments.
	
\end{frame}


\begin{frame}{Refined semantic network:\\semantic vs. logical equivalence}
	\topline
	\justifying
	
	\begin{textitemize}
		\item The prohibition of semantic equivalence does not mean the prohibition of logical equivalence: logical equivalence is useful for compact knowledge representation.
		\item Logically equivalent constructions use different sets of concepts; semantically equivalent ones use the same set.
		\item To minimize logical equivalence, a well-designed hierarchical system of formal ontologies is required.
	\end{textitemize}
	
\end{frame}


\begin{frame}{Refined semantic network:\\network integration}
	\topline
	\justifying
	
	Consequence:
	\begin{textitemize}
		\item Integration of two refined semantic networks comes down to unification followed by conformity with the requirements of absence of synonymy and semantic equivalence.
		\item This ensures correct merging of knowledge without duplicating entities and fragments.
	\end{textitemize}
	
\end{frame}


\begin{frame}{Refined semantic network: Universality and Metainformation}
	\topline
	\justifying
	
	\textbf{Principle 6} (universality):
	\begin{textitemize}
		\item Refined semantic networks should provide representation of any information, including metainformation about the networks themselves.
		\item On a set of networks, a ``be a substructure'' relation is defined that connects networks with their fragments.
		\item Each network is interpreted as a set of signs and a set of connections between them, reflecting the connections between denotates.
	\end{textitemize}
	
\end{frame}


\begin{frame}{Refined semantic network:\\sign denotations}
	\topline
	\justifying
	
	Denotates of signs can be:
	\begin{textitemize}
		\item any described entities --- specific and arbitrary (variables);
		\item real (material) and abstract entities (numbers, points in space, etc.);
		\item external entities and internal entities, which are sets of signs of the same construction.
	\end{textitemize}
	
\end{frame}


\begin{frame}{Refined semantic network:\\linking with files and external formats}
	\topline
	\justifying
	
	\textbf{Principle 7}:
	\begin{textitemize}
		\item The memory of new generation computers must process and store information structures in various formats.
		\item Such constructions can be considered as entities described by a refined semantic network.
		\item Among the network nodes, file nodes appear, the denotations of which (files) are stored in the same memory as the nodes that designate them.
	\end{textitemize}
	
\end{frame}


\begin{frame}{Refined semantic network:\\names and terms}
	\topline
	\justifying
	
	Consequence:
	\begin{textitemize}
		\item The name (term) of an entity described by the same network can be stored in the form of a file described by the network.
		\item The node denoting the name is linked to the node denoting the entity itself.
		\item This allows the terminology of external languages to be aligned with the internal symbols of the network.
	\end{textitemize}
	
\end{frame}


\begin{frame}{Refined semantic network:\\Nonlinearity and Memory}
	\topline
	\justifying
	
	\textbf{Principle 8}:
	\begin{textitemize}
		\item Refined semantic networks are nonlinear symbolic constructions (graph structures).
		\item Nonlinearity complicates the implementation of computer system memory, but simplifies logical organization and associative access.
	\end{textitemize}
	
	\medskip
	Nonlinearity is caused by: the multitude of connections of each entity, the uniqueness of the sign for the entity, the description of all connections as connections between signs.
	
\end{frame}


\begin{frame}{Semantic model of the knowledge base:\\definition}
	\topline
	\justifying
	
	\textbf{Semantic model of a knowledge base} --- a semantic representation of the entire knowledge base of an intelligent computer system in the form of a semantic network belonging to the universal language of semantic networks.
	
	\medskip
	To use semantic networks as a means of knowledge representation it is necessary:
	\begin{textitemize}
		\item consider semantic networks as texts representing different types of knowledge;
		\item to clarify the syntax and semantics of a universal knowledge representation language, the texts of which are semantic networks.
	\end{textitemize}
	
\end{frame}


\begin{frame}{Semantic model of the knowledge base:\\graphodynamic memory}
	\topline
	\justifying
	
	By implementing graphodynamic memory and programming languages oriented towards processing information in such memory, the theoretical advantages of semantic networks can be practically realized in full.
	
\end{frame}


\begin{frame}{Semantic model of the knowledge base:\\advantages (1)}
	\topline
	\justifying
	
	\begin{textitemize}
		\item Providing a high level of associativity in accessing fragments of a knowledge base for a wide range of queries and reducing the computational complexity of searching.
		\item Possibility of effective semantic navigation through the knowledge base with clear 2D/3D visualization of semantic neighborhoods of network elements.
		\item Convenience of knowledge representation for both the intelligent system and users.
	\end{textitemize}
	
\end{frame}


\begin{frame}{Semantic model of the knowledge base:\\advantages (2)}
	\topline
	\justifying
	
	\begin{textitemize}
		\item The ability to consider parts of the model not only as fragments of the external environment, but also as fragments of the knowledge base itself (self-description of the structure).
		\item Explicit selection of knowledge base fragments representing subject areas and ontologies, with an explicit description of their hierarchy and connections (morphisms).
		\item Localization of the scope of knowledge base processing operations through semantic structuring by subject areas.
	\end{textitemize}
	
\end{frame}


\begin{frame}{Semantic model of the knowledge base:\\advantages (3)}
	\topline
	\justifying
	
	\begin{textitemize}
		\item The placement and access to information are determined exclusively by semantic characteristics (sense), and not by the features of memory implementation.
		\item The sense of the information uniquely determines its location in the model and its relationship to the rest of the knowledge base.
	\end{textitemize}
	
\end{frame}


\begin{frame}{Semantic model of the knowledge base:\\compactness of representation}
	\topline
	\justifying
	
	If the volume of representation of an information structure is defined as a pair:
	\begin{textitemize}
		\item number of syntactically elementary fragments;
		\item the number of elements of the alphabet of these fragments,
	\end{textitemize}
	then, among semantically equivalent representations, the most compact is the representation in the form of a refined semantic network.
	
	\medskip
	As the knowledge base expands, the compactness of refined networks increases, since new connections often connect already described entities.
	
\end{frame}


\begin{frame}{Semantic model of the knowledge base:\\text integration}
	\topline
	\justifying
	
	Semantic networks allow for a new understanding of text integration:
	\begin{textitemize}
		\item integration of two semantic networks involves gluing together synonymous elements;
		\item This integration is used when introducing new information into the semantic model of the knowledge base.
	\end{textitemize}
	
\end{frame}


\begin{frame}{Semantic model of the knowledge base:\\knowledge structure analysis}
	\topline
	\justifying
	
	Semantic models of knowledge bases make it possible to:
	\begin{textitemize}
		\item analyze the semantic coherence of the knowledge base by specifying the semantic strength of the connections;
		\item perform knowledge base clustering;
		\item set the metric of semantic distance between characters;
		\item describe correspondences (morphisms) between fragments of the knowledge base (isomorphisms, homomorphisms, analogies and differences).
	\end{textitemize}
	
\end{frame}


\begin{frame}{Semantic model of the knowledge base:\\graph-theoretic algorithms}
	\topline
	\justifying
	
	The use of semantic networks as a knowledge base model allows for the wide application of a range of graph-theoretical algorithms to perform knowledge base processing operations.
	
\end{frame}


\begin{frame}{\\Semantic space}
	\topline
	\justifying
	
	\textbf{Semantic space} (semantic space) ---
	the union of all correct abstract semantic networks belonging to some language of abstract refined semantic networks.
	
	\medskip
	It can be considered as a global (maximal) abstract refined semantic network, including all abstract refined networks of a given language, that is, as an abstract semantic space.
	
\end{frame}
