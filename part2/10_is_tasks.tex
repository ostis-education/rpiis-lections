	\end{SCn}

\begin{frame}{Задача как спецификация действия}
	\topline
	\justifying
	
	\begin{SCn}
		\scnheader{задача}
		\scntext{пояснение}{\textbf{\textit{Задача}}, то есть формальное описание условия некоторой задачи, есть по сути формальная \textit{спецификация} некоторого \textit{действия}, направленного на решение данной \textit{задачи}, достаточная для выполнения данного \textit{действия} каким-либо \textit{субъектом}. В зависимости от конкретного \textit{класса задач} описываться может как внутреннее состояние самой интеллектуальной системы, так и требуемое состояние \textit{внешней среды}.}
	\end{SCn}
\end{frame}

\begin{frame}{Задача и жизненный цикл действия}
	\topline
	\justifying
	
	Каждая \textbf{\textit{задача}} представляет собой спецификацию \textit{действия}, которое либо уже выполнено, либо выполняется сейчас, либо планируется, либо только может быть выполнено. Классификация \textit{задач} может осуществляться по дидактическому признаку в рамках предметной области (задачи на треугольники, задачи на системы уравнений и т.\,п.).
	
\end{frame}

\begin{frame}{Состав задачи: основные компоненты}
	\topline
	\justifying
	
	\vspace{10mm}
	
	Каждая \textit{задача} может включать:
	\begin{textitemize}
		\item принадлежность \textit{действия} частному классу действий и его состояние в жизненном цикле;
		\item описание \textit{цели*} (результата*) действия;
		\item указание \textit{заказчика*} и \textit{исполнителя*} действия;
		\item указание \textit{аргументов действия\scnrolesign}, инструментов и области действия;
		\item описание \textit{декомпозиции действия*} и \textit{последовательности действий*};
		\item условие инициирования действия, моменты начала и завершения, длительность.
	\end{textitemize}
	
\end{frame}

\begin{frame}{Дополнительная информация о задаче}
	\topline
	\justifying
	
	Некоторые \textit{задачи} уточняются контекстом — дополнительной информацией о сущностях, фигурирующих в формулировке задачи. Кроме того, \textit{задача} может включать любую дополнительную информацию о действии:
	\begin{textitemize}
		\item перечень ресурсов и средств (исполнители, сроки, финансы);
		\item ограничения области выполнения действия (например, раздел базы знаний);
		\item ограничения на знания, допустимые при решении задачи.
	\end{textitemize}
	
\end{frame}

\begin{frame}{Классы задач по характеру}
	\topline
	\justifying
	
	С одной стороны, решаемые системой \textit{задачи} делятся на \textit{информационные} и \textit{поведенческие}. С другой стороны, по формулировке выделяют \textit{декларативные формулировки задачи} и \textit{процедурные формулировки задачи}, а также их сочетания.
	
\end{frame}

\begin{frame}{Элементы отношения: декларативная формулировка}

		\begin{textitemize}
			\item связку отношения \textit{цель*} и описание целевой ситуации;
			\item связку отношения \textit{исходная ситуация*} и описание исходной ситуации;

		\begin{textitemize}

			\item \textit{субъектов} действия и их ролей;
			\item \textit{объектов} действия и их ролей;

			\item темпоральных характеристик (сроки, длительность);
			\item приоритета действия;
			\item декомпозиции действия на поддействия вплоть до элементарных.
		\end{textitemize}}
	\end{SCn}

\begin{frame}{Контекст действия: пояснение}
	\topline
	\justifying
	
	\begin{SCn}
		\scnheader{контекст действия*}
		\scntext{пояснение}{Связки отношения \textbf{\textit{контекст действия*}} \textit{действие}, и контекст выполнения данного \textit{действия}, то есть дополнительную информацию о сущностях из описания \textit{цели*}. Контекст задаёт условие задачи («что дано») и определяет знания, которые можно использовать при решении. Даже задачи одного класса действий могут решаться по-разному в разных контекстах.}
	\end{SCn}
	
\end{frame}

\begin{frame}{Форма представления контекста}
	\topline
	\justifying
	
	Контекст может быть представлен не только атомарным фактографическим высказыванием, но и более сложными высказываниями, например:
	\begin{textitemize}
		\item определение множества, используемого в описании \textit{цели*};
		\item утверждение, учёт которого может быть полезен при решении задач.
	\end{textitemize}
	
\end{frame}

\begin{frame}{Протокол: пояснение}
	\topline
	\justifying
	
	\begin{SCn}
		\scnheader{протокол}
		\scntext{пояснение}{Каждый \textbf{\textit{протокол}} — это \textit{спецификация} \textit{действия}, процесс выполнения которого описывается, включая поддействия, оказавшиеся впоследствии нецелесообразными. Предполагается, что знак действия относится к множеству прошлых сущностей. В отличие от \textit{плана}, протокол всегда формируется по факту выполнения действия.}
	\end{SCn}
\end{frame}

\begin{frame}{Декомпозиция и стратегия решения задач}
	\topline
	\justifying
	
	Каждое \textit{действие} может иметь несколько вариантов декомпозиции в зависимости от набора элементарных действий, доступных системе субъектов. Принцип, по которому осуществляется декомпозиция в разных подходах к решению задач, называется \textit{стратегией решения задач}.
	
\end{frame}

	\topline
	\justifying
	
	\begin{SCn}
		\scnheader{\scnkeyword{исходная ситуация}*}
		\scnidtf{исходная ситуация, соответствующая заданному действию*}

\begin{frame}{Проблемы традиционного поиска}
	\topline
	\justifying
	
	\begin{textitemize}
		\item Традиционные информационно-поисковые системы нередко возвращают большое количество нерелевантных документов, поэтому ключевым критерием качества поиска становится релевантность.
		\item Простая автоматизация поиска и многокритериальная классификация информации сами по себе не обеспечили существенного роста релевантности и комфортности поиска.
		\item Существенная проблема состоит и в том, что документы часто выдаются как беспорядочная выборка, а не как упорядоченная по степени важности последовательность.
	\end{textitemize}
\end{frame}

\begin{frame}{Технологии интеллектуального поиска}
	\topline
	\justifying
	
	\begin{textitemize}
		\item Для интеллектуализации поиска применяются системы управления знаниями, так как они помогают организовать информацию и упростить решение задач поиска.
		\item В качестве основы интеллектуальных ИПС предлагаются онтологический и мультиагентный подходы.
		\item Важную роль играют и другие методы искусственного интеллекта, включая нечеткую логику и машинное обучение, позволяющие работать с неопределённостью и приблизительными рассуждениями.
	\end{textitemize}
\end{frame}

\begin{frame}{Персонализация поиска}
	\topline
	\justifying
	
	\begin{textitemize}
		\item Современные поисковые системы движутся от общезначимой релевантности к индивидуальной релевантности, то есть к учёту особенностей конкретного пользователя.
		\item Под персонализацией понимается выдача результатов с учётом информационных потребностей, интересов, географического положения, возраста и других характеристик пользователя.
		\item Для оценки предпочтений используются коллаборативная фильтрация, эвристическое моделирование и поведенческий таргетинг.
	\end{textitemize}
\end{frame}

\begin{frame}{Архитектура интеллектуальной ИПС}
	\topline
	\justifying
	
	\begin{textitemize}
		\item Перспективной считается мультиагентная архитектура, в которой взаимодействуют интеллектуальные агенты на серверной и клиентской сторонах.
		\item Такие агенты могут выполнять традиционные функции поисковой машины, дополняя их мониторингом предпочтений пользователя и интеллектуальной обработкой запросов.
		\item Собранные сведения о запросах и интересах пользователя могут помещаться в библиотеку прецедентов и использоваться для последующей математической и статистической обработки.
	\end{textitemize}
\end{frame}

\begin{frame}{Эффект интеллектуализации поиска}
	\topline
	\justifying
	
	\begin{textitemize}
		\item Интеллектуальные и персонализированные механизмы поиска должны применяться совместно с традиционными поисковыми средствами, поскольку именно в комплексе их эффективность возрастает.
		\item Главный практический эффект интеллектуализации состоит в приближении к решению проблемы выбора действительно релевантной информации из больших массивов данных.
		\item Тем самым интеллектуальный поиск становится не просто средством выдачи документов, а инструментом поддержки пользователя в извлечении знаний и принятии решений.
	\end{textitemize}
\end{frame}

\begin{frame}{Сущность трудноформализуемых задач}
	\topline
	\justifying
	
	\begin{textitemize}
		\item Для таких задач трудно или невозможно построить точную математическую постановку.
		\item Даже если формальная постановка существует, пространство поиска решения может быть слишком большим для практического перебора.
		\item В подобных задачах часто отсутствует единый универсальный алгоритм решения.
	\end{textitemize}
\end{frame}

\begin{frame}{Основные признаки}
	\topline
	\justifying
	
	\begin{textitemize}
		\item Неполнота, неточность или недостоверность исходных данных.
		\item Неоднозначность условий, ограничений и самого понятия решения.
		\item Наличие противоречивых или несогласованных критериев оценки результата.
		\item Зависимость решения от контекста, опыта и интерпретации человека.
	\end{textitemize}
\end{frame}

\begin{frame}{Почему постановка нечёткая}
	\topline
	\justifying
	
	\begin{textitemize}
		\item Цели задачи могут быть сформулированы расплывчато или изменяться в процессе решения.
		\item Не всегда известны все существенные факторы и взаимосвязи между ними.
		\item Часть параметров имеет качественный, а не количественный характер.
		\item В задаче могут одновременно присутствовать неопределённость, субъективность и изменяющиеся условия.
	\end{textitemize}
\end{frame}

\begin{frame}{Основные стратегии решения задач}
	\topline
	\justifying
	
	\vspace{10mm}
	
	Если для достижения информационной цели текущего контекста базы знаний недостаточно, возможны три основные стратегии:
	
	\begin{textitemize}
		\item \textbf{Декомпозиция цели:} исходная цель сводится к системе подцелей и подзадач, пока не будут получены такие подцели, для которых уже существует применимый метод решения.
		\item \textbf{Генерация новых знаний:} в семантической окрестности цели порождаются новые знания с помощью доступных методов, чтобы получить контекст, достаточный для достижения исходной цели.
		\item \textbf{Комбинированная стратегия:} совместное использование декомпозиции и генерации новых знаний.
	\end{textitemize}
\end{frame}
