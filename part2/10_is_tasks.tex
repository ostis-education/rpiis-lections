\begin{frame}{\\Задачи на анализ}
	\topline
	\justifying
	
	\begin{SCn}
		\scnheader{задачи на анализ}
		\scnidtf{задачи, которые необходимо решить для ответов на вопросы «Как связаны два указанных понятия», «В чем различие между указанными объектами» и т.д}
        \end{SCn}
        Поиск ответа на такого рода вопросы часто ограничивается простым поиском в базе знаний, однако не менее часто возникают ситуации, когда для ответа необходимо сгенерировать новые знания на основе имеющихся.
\end{frame}

\begin{frame}{\\Поведенческая задача}
	\topline
	\justifying
	
	\begin{SCn}
		\scnheader{поведенческая задача}
		\scntext{пояснение}{Предполагает наличие некоторой среды, в которой находится агент, призванный решить данную задачу.}
		\begin{scnrelfromset}{примеры}
			\scnitem{физические задачи}
			\scnitem{геометрические задачи}
			\scnitem{химические задачи}
		\end{scnrelfromset}
	\end{SCn}

	Задача агента состоит в том, чтобы преобразовать окружающую среду из начального состояния в конечное, используя при этом допустимые операции преобразования из имеющегося множества
\end{frame}

		\scntext{пояснение}{Две сущности будем считать связанными в том и только в том случае, если существует маршрут, соединяющий указанные две сущности, в состав которого входят связки, принадлежащие в общем случаем разным отношениям}
		\scntext{пояснение}{Здесь принципиально важным является учет \textit{семантической силы связей} между сущностями, которая определяется \textit{семантической силой отношений}, которым принадлежат связки, входящие в состав связей (маршрутов) между сущностями.}
	\end{SCn}

            \scnfileitem{выбор информации, осуществляемый человеком, нередко не является рациональным и строго последовательным, что сущест­венно осложняет поиск информации}
            \scnfileitem{пользователь при поиске информации обычно не строго определяет цель поиска, т. е. использует нечетко определенные понятия}
    \end{scnrelfromset}
